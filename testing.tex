\section*{Directory Tree}

% Directory tree example
\begin{forest}
  for tree={
    font=\ttfamily,
    grow'=0,
    child anchor=west,
    parent anchor=south,
    anchor=west,
    calign=first,
    edge path={
      \noexpand\path [draw, \forestoption{edge}]
      (!u.south west) +(7.5pt,0) |- (.child anchor) \forestoption{edge label};
    },
    before typesetting nodes={
      if n=1
        {insert before={[,phantom]}}
        {}
    },
    fit=band,
    before computing xy={l=15pt},
  }
[project/
  [src/
    [main.py]
    [config.py]
  ]
  [data/
    [data1.csv]
    [data2.csv]
  ]
  [docs/
    [README.md]
  ]
  [output/
    [results.txt]
  ]
]
\end{forest}
